== NgFx

Reactive hooks for Angular.

[source,typescript]
----
@Component({
    selector: "app-root",
    template: `
        <div>Count: {{ count }}</div>
    `,
})
export class AppComponent extends Connectable {
    @Input()
    count = 0

    ngOnConnect() {
        effect(() =>
            interval(1000).subscribe(() =>
                this.count += 1
            )
        )
    }
}
----
